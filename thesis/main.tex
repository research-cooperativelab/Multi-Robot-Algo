% sectionlinesformat is called for every \section / \subsection heading.
% #1 = level name ("section", "subsection", …)
% #2 = indent width (0pt when secnumdepth=0)
% #3 = label/number (empty when secnumdepth=0)
% #4 = heading text
%
% CSULB rule: first-level (section) → centered + underlined;
%             second-level (subsection) → left margin + underlined.
\makeatletter
\renewcommand\sectionlinesformat[4]{%
  \Ifstr{#1}{section}{%
    {\centering\uline{#4}\par}%   % first-level: centered + underlined
  }{%
    \uline{#4}%                   % second-level and below: left + underlined
  }%
}
\makeatother
